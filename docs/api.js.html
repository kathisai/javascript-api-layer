let singletonAxiosInstance = require('./server.js')
let LRU = require('./LRU.js');
let lruCache = new LRU(100); // Cache limit is 10 items
/**
 * This class act as base class for JavaScript API.
 * All network methods will be added to this class.
 * User has to acces only this class from outside
 */
class Api {
    constructor() {
        this.server = singletonAxiosInstance;
    }

    /**
     * Get all users from API
     * @param  params 
     * @param  s  Sucess function  
     * @param  e  Error function
     */
    getUsers(params, s = success, e = error) {
        this.server.get('/users/', params)
            .then(
                data => {
                    var cache = lruCache.read('users');
                    if (cache) {
                        console.log('from cache', cache);
                        s(cache);
                    } else {
                        lruCache.write('users', data.data);
                        console.log('stored in cache');
                        s(data.data);
                    }
                }
            )
            .catch(err => {
                e(err)
            })
    }

    /**
     * Get specific user
     * @param  id  unique id for users
     * @param  params 
     * @param s Sucess function 
     * @param  e  Error function
     */
    getUser(id, params, s = success, e = error) {
        this.server.get(`/users/${id}`, params)
            .then(data => s(data.data))
            .catch(err => e(err))
    }

    /**
     * Post user details 
     * @param  data 
     * @param  params 
     * @param  s Sucess function 
     * @param  e  Error function
     */
    postUser(data, params, s = success, e = error) {
        this.server.post(`/users/`, data, params)
            .then(data => s(data.data))
            .catch(err => e(err))
    }

    /**
     * Deleted Users details
     * @param  id unique id for users
     * @param  params 
     * @param  s Sucess function
     * @param  e  Error function
     */
    deleteUser(id, params, s = success, e = error) {
        this.server.delete(`/users/${id}`, id, params)
            .then(data => s(data.data))
            .catch(err => e(err))
    }


}
/**
 * Defalt sucess function log
 * @param  data 
 */
const success = (data) => {
    console.log("Data : ", data);
}

/**
 * Default error function log
 * @param  err 
 */
const error = (err) => {
    console.log("Error : ", err);
}


const api = new Api();


module.exports = api;
