let constants = require('./constants');
let axiosRetry = require("axios-retry");
let axios = require("axios");

let instance = null;
/**
 * Provide Axios reference to outside classes and also have generic
 * network methods
 */
class Server {
  constructor() {
    this.client = axios.create({
      baseURL: constants.PREVIEW_URL
    });
    axiosRetry(this.client, { retries: 3 });

    if (instance) {
      return instance;
    } else {
      instance = this;
      return instance;
    }
  }

  /**
   * Generic netwrok get method
   * @param  path 
   * @param params 
   */
  get(path, params) {
    return this.client.get(path, {
      params: params
    });
  }

  /**
   * Generic netwrok Post method
   * @param  path 
   * @param  data 
   * @param  params 
   */
  post(path, data, params) {
    return this.client.post(path, data, {
      params: params
    });
  }

  /**
   *  Generic netwrok Delete method
   * @param  path 
   * @param  id 
   * @param  params 
   */
  delete(path, id, params) {
    return this.client.delete(path, {
      params: {
        id: id
      }
    });
  }
}


let singletonAxiosInstance = new Server();

module.exports = singletonAxiosInstance;
